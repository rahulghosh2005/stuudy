From src/hooks/useTimer.ts (Plan 02):
```typescript
export type TimerMode = 'stopwatch' | 'pomodoro';
export type TimerStatus = 'idle' | 'running' | 'stopped';
export interface TimerState {
  mode: TimerMode;
  status: TimerStatus;
  startTimestamp: number | null;  // Date.now() at Start — use as startMs for addSession
  elapsed: number;                // milliseconds — current phase only
  totalElapsed: number;           // milliseconds — sum of all completed Pomodoro phases
  pomodoroPhase: PomodoroPhase;
  pomodoroWorkMs: number;
  pomodoroBreakMs: number;
}
export function useTimer(): { state: TimerState; start: () => void; stop: () => void; reset: () => void; setMode: (mode: TimerMode) => void };
export function formatElapsed(ms: number): string;
```

CRITICAL — durationMs calculation:
  durationMs = state.totalElapsed + state.elapsed

  state.elapsed holds only the CURRENT phase's elapsed time. state.totalElapsed holds the sum
  of all COMPLETED phases. Passing state.elapsed alone would record only the last phase, not the
  full session duration. This is required by TIMR-06. For stopwatch sessions, totalElapsed is always
  0 (no phase transitions), so the formula collapses to state.elapsed — correct in both cases.

    ```typescript
    async function handleSave(notes: string) {
      if (!user || state.startTimestamp === null) return;
      try {
        await addSession(user.uid, {
          subject: selectedSubject?.name ?? null,
          subjectId: selectedSubject?.id ?? null,
          // CRITICAL: sum of all completed phases (totalElapsed) plus current phase (elapsed).
          // Using state.elapsed alone would record only the last Pomodoro phase, not the full
          // session duration. For stopwatch, totalElapsed is 0, so this correctly equals elapsed.
          durationMs: state.totalElapsed + state.elapsed,
          startMs: state.startTimestamp,  // Date.now() captured at Start — useTimer.state.startTimestamp
          notes,
        });
        reset();               // reset timer to idle (also resets totalElapsed to 0)
        setSelectedSubject(null);
        setShowToast(true);
        setTimeout(() => setShowToast(false), 3000);
      } catch (err) {
        setSaveError('Failed to save session. Try again.');
        console.error(err);
      }
    }
    ```
