**Create src/components/BottomTabBar.tsx** (from RESEARCH.md Code Examples — NavLink pattern):

```typescript
import { NavLink } from 'react-router-dom';

export function BottomTabBar() {
  const tabStyle = (isActive: boolean): React.CSSProperties => ({
    flex: 1,
    display: 'flex',
    flexDirection: 'column',
    alignItems: 'center',
    padding: '10px 0',
    color: isActive ? '#fc4c02' : '#555',
    textDecoration: 'none',
    fontSize: 11,
    fontWeight: isActive ? 700 : 400,
    borderTop: `2px solid ${isActive ? '#fc4c02' : 'transparent'}`,
  });

  return (
    <nav style={{
      position: 'fixed',
      bottom: 0,
      left: 0,
      right: 0,
      background: '#0a0a0a',
      display: 'flex',
      borderTop: '1px solid #1a1a1a',
      zIndex: 100,
      paddingBottom: 'env(safe-area-inset-bottom)', // iOS safe area
    }}>
      <NavLink to="/" end style={({ isActive }) => tabStyle(isActive)}>
        Timer
      </NavLink>
      <NavLink to="/stats" style={({ isActive }) => tabStyle(isActive)}>
        Stats
      </NavLink>
      <NavLink to="/profile" style={({ isActive }) => tabStyle(isActive)}>
        Profile
      </NavLink>
    </nav>
  );
}
```

Note: `to="/" end` prevents `/` from being active on `/stats` — the `end` prop requires exact match.

**Create src/pages/StatsPage.tsx:**

This page:
1. Fetches the signed-in user's UserProfile from Firestore (for timezone + goal fields)
2. Uses `useSessions` for the selected range (starts at '1W')
3. Uses `useStats` to get chart data, breakdown, heatmap, total
4. Uses `useStreak` for streak counts (also writes back to Firestore)
5. Renders sections top to bottom: chart → heatmap → streak + total summary → subject breakdown → goals

**Data flow:**

```typescript
const { user } = useAuth();
const [profile, setProfile] = useState<UserProfile | null>(null);
const [range, setRange] = useState<TimeRange>('1W');
const [subjectId, setSubjectId] = useState<string | null>(null);

// Fetch profile (contains timezone + goal fields)
useEffect(() => {
  if (!user?.uid) return;
  getDoc(doc(db, 'users', user.uid))
    .then(snap => { if (snap.exists()) setProfile(snap.data() as UserProfile); })
    .catch(() => {});
}, [user?.uid]);

const timezone = profile?.timezone ?? Intl.DateTimeFormat().resolvedOptions().timeZone;
const { sessions, loading } = useSessions(user?.uid ?? '', range, timezone);
const { chartData, breakdown, heatmap, total } = useStats(sessions, range, timezone, subjectId);
const { currentStreak, longestStreak } = useStreak(sessions, timezone, user?.uid ?? '');
```

Updated structure — add a `LayoutWithNav` wrapper that renders the Outlet plus BottomTabBar:

```typescript
import { BrowserRouter, Routes, Route, Outlet } from 'react-router-dom';
import { AuthProvider } from './contexts/AuthContext';
import { ProtectedRoute } from './components/ProtectedRoute';
import { LoginPage } from './pages/LoginPage';
import { ProfilePage } from './pages/ProfilePage';
import { TimerPage } from './pages/TimerPage';
import { StatsPage } from './pages/StatsPage';
import { BottomTabBar } from './components/BottomTabBar';

function LayoutWithNav() {
  return (
    <>
      <Outlet />
      <BottomTabBar />
    </>
  );
}

export default function App() {
  return (
    <AuthProvider>
      <BrowserRouter>
        <Routes>
          <Route path="/login" element={<LoginPage />} />
          <Route element={<ProtectedRoute />}>
            <Route element={<LayoutWithNav />}>
              <Route path="/" element={<TimerPage />} />
              <Route path="/stats" element={<StatsPage />} />
              <Route path="/profile" element={<ProfilePage />} />
              <Route path="/profile/:uid" element={<ProfilePage />} />
            </Route>
          </Route>
        </Routes>
      </BrowserRouter>
    </AuthProvider>
  );
}
```

This nests `LayoutWithNav` inside `ProtectedRoute` — so BottomTabBar is only rendered for authenticated users. LoginPage does NOT get the tab bar.
