**Custom tooltip component (defined above StudyChart):**
```typescript
function CustomTooltip({ active, payload, label }: TooltipProps<number, string>) {
  if (!active || !payload || payload.length === 0) return null;
  const minutes = payload[0].value as number;
  const hours = Math.floor(minutes / 60);
  const mins = minutes % 60;
  return (
    <div style={{ background: '#1a1a1a', border: '1px solid #333', borderRadius: 6, padding: '8px 12px' }}>
      <p style={{ color: '#888', margin: 0, fontSize: 12 }}>{label}</p>
      <p style={{ color: '#fc4c02', margin: '4px 0 0', fontWeight: 700 }}>
        {hours > 0 ? `${hours}h ${mins}m` : `${mins}m`}
      </p>
    </div>
  );
}
```

`classForValue` function:
```typescript
(value) => {
  if (!value || value.count === 0) return 'color-scale-0';
  if (value.count < 30) return 'color-scale-1';
  if (value.count < 60) return 'color-scale-2';
  if (value.count < 120) return 'color-scale-3';
  return 'color-scale-4';
}

**View toggle** (year vs month — like a stock chart period selector):

**Tooltip:** Simple controlled tooltip using `onMouseOver`/`onMouseLeave` on the heatmap wrapper (NOT a third-party tooltip library). Track `tooltip: { date: string; count: number; visible: boolean }` in state. On `onMouseOver` of the heatmap container, show a `<div>` absolutely positioned. This avoids adding a tooltip library.

}
```

Full component structure:
```tsx
export function StudyHeatmap({ values }: StudyHeatmapProps) {
  const [view, setView] = useState<'year' | 'month'>('year');
  const [selectedMonth, setSelectedMonth] = useState(new Date());

  const startDate = view === 'year'
    ? subYears(new Date(), 1)
    : startOfMonth(selectedMonth);
  const endDate = view === 'year'
    ? new Date()
    : endOfMonth(selectedMonth);

  return (
    <div style={{ background: '#1a1a1a', borderRadius: 12, padding: '16px' }}>
      <style>{HEATMAP_STYLE}</style>
      {/* View toggle + month navigation */}
      ...
      <CalendarHeatmap
        startDate={startDate}
        endDate={endDate}
        values={values}
        classForValue={classForValue}
        tooltipDataAttrs={...}
        showWeekdayLabels
      />
    </div>
  );
}
```
